Gestion des images

# Télécharger une image depuis Docker Hub
docker pull <nom_image>:<tag>
# Lister les images disponibles localement
docker images
# Construire une image à partir d'un Dockerfile
docker build -t <nom_image>:<tag> <chemin_dockerfile>
# Supprimer une image
docker rmi <nom_image>:<tag>
# Nettoyer les images non utilisées
docker image prune
# Inspecter une image
docker image inspect <nom_image>:<tag>
# Pousser une image vers un registry
docker push <nom_image>:<tag>
# Sauvegarder une image dans un fichier tar
docker save -o <fichier_sortie>.tar <nom_image>:<tag>
# Charger une image depuis un fichier tar
docker load -i <fichier_entree>.tar

Gestion des conteneurs

# Créer et démarrer un conteneur
docker run <options> <nom_image>:<tag>
# Options courantes pour docker run
#   -d : Exécuter en arrière-plan (mode détaché)
#   -p <port_hote>:<port_conteneur> : Mapper un port
#   -v <volume_hote>:<volume_conteneur> : Monter un volume
#   --name <nom> : Donner un nom au conteneur
#   --rm : Supprimer automatiquement le conteneur à l'arrêt
#   -e <VAR>=<valeur> : Définir une variable d'environnement
# Lister les conteneurs en cours d'exécution
docker ps
# Lister tous les conteneurs (y compris ceux arrêtés)
docker ps -a
# Démarrer un conteneur arrêté
docker start <id_conteneur|nom_conteneur>
# Arrêter un conteneur
docker stop <id_conteneur|nom_conteneur>
# Redémarrer un conteneur
docker restart <id_conteneur|nom_conteneur>
# Supprimer un conteneur
docker rm <id_conteneur|nom_conteneur>
# Exécuter une commande dans un conteneur en cours d'exécution
docker exec -it <id_conteneur|nom_conteneur> <commande>
# Accéder au shell d'un conteneur
docker exec -it <id_conteneur|nom_conteneur> /bin/bash
# Voir les logs d'un conteneur
docker logs <id_conteneur|nom_conteneur>
# Suivre les logs en temps réel
docker logs -f <id_conteneur|nom_conteneur>
# Inspecter un conteneur
docker inspect <id_conteneur|nom_conteneur>
# Voir les statistiques d'utilisation des ressources
docker stats <id_conteneur|nom_conteneur>
# Copier des fichiers entre l'hôte et le conteneur
docker cp <chemin_source> <id_conteneur|nom_conteneur>:<chemin_destination>
docker cp <id_conteneur|nom_conteneur>:<chemin_source> <chemin_destination>

Gestion des volumes

# Créer un volume
docker volume create <nom_volume>
# Lister les volumes
docker volume ls
# Inspecter un volume
docker volume inspect <nom_volume>
# Supprimer un volume
docker volume rm <nom_volume>
# Nettoyer les volumes non utilisés
docker volume prune

Gestion des réseaux

# Créer un réseau
docker network create <options> <nom_reseau>
# Lister les réseaux
docker network ls
# Inspecter un réseau
docker network inspect <nom_reseau>
# Connecter un conteneur à un réseau
docker network connect <nom_reseau> <id_conteneur|nom_conteneur>
# Déconnecter un conteneur d'un réseau
docker network disconnect <nom_reseau> <id_conteneur|nom_conteneur>
# Supprimer un réseau
docker network rm <nom_reseau>
# Nettoyer les réseaux non utilisés
docker network prune

Docker Compose

# Démarrer les services définis dans docker-compose.yml
docker-compose up
# Démarrer les services en arrière-plan
docker-compose up -d
# Arrêter les services
docker-compose down
# Arrêter les services et supprimer les volumes
docker-compose down -v
# Voir les logs des services
docker-compose logs
# Suivre les logs en temps réel
docker-compose logs -f
# Exécuter une commande dans un service
docker-compose exec <service> <commande>
# Voir l'état des services
docker-compose ps
# Reconstruire les images des services
docker-compose build
# Démarrer un service spécifique
docker-compose up <service>
# Mettre à l'échelle un service
docker-compose up -d --scale <service>=<nombre_instances>

Informations système

# Afficher les informations système Docker
docker info
# Afficher la version de Docker
docker version
# Vérifier l'état du daemon Docker
docker system info
# Afficher l'utilisation des ressources
docker system df
# Nettoyer les ressources non utilisées (images, conteneurs, volumes, réseaux)
docker system prune
# Nettoyer tout, y compris les volumes non utilisés
docker system prune -a --volumes

Sécurité

# Scanner une image pour les vulnérabilités (nécessite Docker Desktop)
docker scan <nom_image>:<tag>
# Vérifier les autorisations d'un conteneur
docker exec <id_conteneur|nom_conteneur> ls -la /path/to/check
# Exécuter un conteneur en mode privilégié (à éviter en production)
docker run --privileged <nom_image>:<tag>
# Limiter les ressources CPU d'un conteneur
docker run --cpus=".5" <nom_image>:<tag>
# Limiter la mémoire d'un conteneur
docker run --memory="512m" <nom_image>:<tag>
